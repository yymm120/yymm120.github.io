                                                                                                                                                                      sort 与 sorted 区别：

                                                                                                                                                                      sort 是应用在 list 上的方法，sorted 可以对所有可迭代的对象进行排序操作。

                                                                                                                                                                      list 的 sort 方法返回的是对已经存在的列表进行操作，无返回值，而内建函数 sorted 方法返回的是一个新的 list，而不是在原来的基础上进行的操作。

                                                                                                                                                                      sorted 语法：

                                                                                                                                                                      sorted(iterable, cmp=None, key=None, reverse=False)

                                                                                                                                                                      参数说明：

                                                                                                                                                                      • iterable -- 可迭代对象。
                                                                                                                                                                      • cmp -- 比较的函数，这个具有两个参数，参数的值都是从可迭代对象中取出，此函数必须遵守的规则为，大于则返回1，小于则返回-1，等于则返回0。
                                                                                                                                                                      • key -- 主要是用来进行比较的元素，只有一个参数，具体的函数的参数就是取自于可迭代对象中，指定可迭代对象中的一个元素来进行排序。
                                                                                                                                                                      • reverse -- 排序规则，reverse = True 降序 ， reverse = False 升序（默认）。

                                                                                                                                                                      示例：

                                                                                                                                                                      # 对字典排序

# 1. 哨兵节点——哨兵节点在树和链表中被广泛用作伪头、伪尾。通常不保存任何数据
class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None

class MyLinkedList:
    def __init__(self):
        self.size = 0
        self.head = ListNode(0) 
        
    def get(self, index: int) -> int:
        if index < 0 or index >= self.size:
            return -1
        
        curr = self.head
        for _ in range(index + 1):
            curr = curr.next
        return curr.val
            
    def addAtHead(self, val: int) -> None:
        self.addAtIndex(0, val)
        
    def addAtTail(self, val: int) -> None:
        self.addAtIndex(self.size, val)
        
    def addAtIndex(self, index: int, val: int) -> None:
        if index > self.size:
            return
        
        if index < 0:
            index = 0
        
        self.size += 1
        pred = self.head
        for _ in range(index):
            pred = pred.next
            
        to_add = ListNode(val)
        to_add.next = pred.next
        pred.next = to_add
        
    def deleteAtIndex(self, index: int) -> None:
        if index < 0 or index >= self.size:
            return
        
        self.size -= 1
        pred = self.head
        for _ in range(index):
            pred = pred.next
            
        pred.next = pred.next.next

from winreg import *

# tweak as necessary
version = sys.version[:3]
installpath = sys.prefix

regpath = "SOFTWARE\\Python\\Pythoncore\\%s\\" % (version)
installkey = "InstallPath"
pythonkey = "PythonPath"
pythonpath = "%s;%s\\Lib\\;%s\\DLLs\\" % (installpath, installpath, installpath)


def RegisterPy():
    try:
        reg = OpenKey(HKEY_CURRENT_USER, regpath)
    except EnvironmentError as e:
        try:
            reg = CreateKey(HKEY_CURRENT_USER, regpath)
            SetValue(reg, installkey, REG_SZ, installpath)
            SetValue(reg, pythonkey, REG_SZ, pythonpath)
            CloseKey(reg)
        except:
            print("*** Unable to register!")
            return
        print("--- Python", version, "is now registered!")
        return
    if (
        QueryValue(reg, installkey) == installpath
        and QueryValue(reg, pythonkey) == pythonpath
    ):
        CloseKey(reg)
        print("=== Python", version, "is already registered!")
        return
    CloseKey(reg)
    print("*** Unable to register!")
    print("*** You probably have another Python installation!")


if __name__ == "__main__":
    RegisterPy()

                                                                                                                                                                      闭包：函数内部嵌套函数，内层函数称为闭包。

                                                                                                                                                                      该模块提供了读写Mac OS使用的__属性列表文件__[^1]的接口。支持二进制和XML plist文件。

                                                                                                                                                                      [^1]:属性列表文件(.plist) 格式，是一种简单的序列化，支持字典、列表、数字、字符串。顶级对象是字典。

                                                                                                                                                                      dump() 和 load() 操作文件

                                                                                                                                                                      dumps() 和 loads() 操作字节对象中的plist数据。

                                                                                                                                                                      使用：

                                                                                                                                                                      load() 从文件中载入

                                                                                                                                                                      plistlib.load(*fp, fmt=None, use_builtin_types=True, dict_type=dict)

                                                                                                                                                                      示例：

                                                                                                                                                                      plistlib.load('./xx.plist', fmt=FMT_BINARY)

                                                                                                                                                                      dump() 写入文件中

                                                                                                                                                                      dump(*value, fmt=None, sort_key=True, skipkeys=False)

                                                                                                                                                                      示例：

                                                                                                                                                                      plistlib.dump(data, './xx.plist')

                                                                                                                                                                      loads() 从bytes对象中载入

                                                                                                                                                                      plistlib. loads(* data *，***，* fmt = None *，* use_builtin_types = True *，* dict_type = dict *)

                                                                                                                                                                      示例：

                                                                                                                                                                      plistlib.loads(data_byte, fmt=None)

                                                                                                                                                                      dumps() 将数据转为plist格式的字节对象。

                                                                                                                                                                      plistlib. dumps(* value *，***，* fmt = FMT_XML *，* sort_keys = True *，* skipkeys = False *)

                                                                                                                                                                      示例：

                                                                                                                                                                      data_byte = plistlib.dumps(data, fmt=FMT_BINARY)

                                                                                                                                                                      常量

                                                                                                                                                                      plistlib.FMT_XML	plist文件的XML格式
plistlib.FMT_BINARY		plist文件的二进制格式

                                                                                                                                                                      pandoc -o output.html input.txt

                                                                                                                                                                      iconv -t utf-8 input.txt | pandoc | iconv -f utf-8

                                                                                                                                                                      创建窗口

                                                                                                                                                                      from tkinter import Tk
# 1. 显示窗口
my_window = Tk()
# 2. 标题
my_window.title("我的窗口")

# 3. 设置窗口的宽、高
width = 240
height = 480
align_str = '%dx%d' % (width, height)
my_window.geometry(align_str)

# 4. 设置窗口是否可以缩放拉伸, True表示可以缩放
my_window.resizable(width=False, height=False)
# 5. 设置窗口居中显示
# 先获取整个窗口的大小
screen_width, screen_height = my_window.maxsize()
# 然后再设置geometry()
my_window.geometry("%dx%d+%d+%d" % (width, height, (screen_width-width)/2, (screen_height-height)/2))

# 6. loop循环
my_window.mainloop()
