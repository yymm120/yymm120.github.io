                                                                                                                                                                      1. workspace设置

                                                                                                                                                                      基本思想是，我们将创建一个包含不同 crate 的单体repo，这些 crate 将被编译在一起。

                                                                                                                                                                      ├── api            # Rust API code
│   ├── lib        # Actix-Web API code
│   └── shuttle    # Shuttle project
├── front          # Dioxus front-end code
├── shared         # Common code shared between the API and the front-end
└── README.md      # Workshop instructions and guidance

                                                                                                                                                                      1.1 创建workspace

                                                                                                                                                                      在一个新的目录中，创建Cargo.toml，并添加如下workspace

                                                                                                                                                                      [workspace]
members = [
    "api/lib",
    "api/shuttle",
    "shared"
]

                                                                                                                                                                      front会在后面创建，不用担心

                                                                                                                                                                      1.2 初始化git仓库

                                                                                                                                                                      git init

                                                                                                                                                                      1.3 创建crates

                                                                                                                                                                      • lib
                                                                                                                                                                      • shuttle
                                                                                                                                                                      • shared

                                                                                                                                                                      创建lib crate 

                                                                                                                                                                      cargo new api/lib --name api-lib --lib --vcs none

                                                                                                                                                                      --vcs none用于高数Cargo不要初始化新的git存储库

                                                                                                                                                                      此命令执行后可能会出现一个错误，不用担心。后文将会修复它。

                                                                                                                                                                      创建shuttle crate 

                                                                                                                                                                       cargo shuttle init api/shuttle -t axum --name api-shuttle-template

                                                                                                                                                                      请提前下载cargo-shuttle，cargo install cargo-shuttle

                                                                                                                                                                      创建shared crate 

                                                                                                                                                                      cargo new shared  --lib

                                                                                                                                                                      这里没有使用--name选项，这意味着包的名称将从文件名中推断出来。

                                                                                                                                                                      cargo build

                                                                                                                                                                      6. 连接数据库

                                                                                                                                                                      在Axum中应用数据库，要求数据库必须实现Clone，才能在axum::State<S>中传递。有两种方式可以做到这一点，添加#[derive(Clone)]宏或者std::sync::Arc引用。

                                                                                                                                                                      #[derive(Clone)]
struct AppState {
    pool: PgPool,
}
let state = AppState { pool };

async fn version(
    State(state): State<AppState>
) -> String {
    
}

                                                                                                                                                                      or

                                                                                                                                                                      use std::sync::Arc;

struct AppState {
    pool: PgPool,
}

let state = Arc::new(AppState { db });

async fn version(
	State(state): State<std::sync::Arc<AppState>>,
) -> String {
    
}

                                                                                                                                                                      #[derive(Clone)]和std::sync::Arc;的区别在于，后者在提取参数时不是State<AppState>而是State<Arc<AppState>>

                                                                                                                                                                      version()方法用于获取数据库中的数据。这里sqlx::query_scalar("SELECT version()")用于获取数据库的版本。

                                                                                                                                                                      async fn version(
    State(state): State<std::sync::Arc<AppState>>,
) -> String {
       let result: Result<String, sqlx::Error> = sqlx::query_scalar("SELECT version()")
        .fetch_one(&state.pool.clone())
        .await;

    match result {
        Ok(version) => version,
        Err(e) => format!("Error: {:?}", e),
    }
}

                                                                                                                                                                      完整代码如下：

                                                                                                                                                                      use axum::{routing::get, Router, extract::State};
use sqlx::Executor;
use shuttle_runtime::CustomError;
use sqlx::PgPool;

async fn hello_world() -> &'static str {
    "Hello, world!"
}

// #[derive(Clone)]
struct AppState {
    pool: PgPool,
}

async fn version(
    State(state): State<std::sync::Arc<AppState>>,
) -> String {
       let result: Result<String, sqlx::Error> = sqlx::query_scalar("SELECT version()")
        .fetch_one(&state.pool.clone())
        .await;

    match result {
        Ok(version) => version,
        Err(e) => format!("Error: {:?}", e),
    }
}

#[shuttle_runtime::main]
async fn main(
    #[shuttle_shared_db::Postgres] pool: sqlx::PgPool,
    // #[shuttle_secrets::Secrets] secrets: shuttle_secrets::SecretStore,
) -> shuttle_axum::ShuttleAxum {
    pool.execute(include_str!("../../db/schema.sql"))
        .await
        .map_err(CustomError::new)?;

    let state = std::sync::Arc::new(AppState {pool});
    let router = Router::new()
        .route("/", get(hello_world))
        .route("/version", get(version))
        .with_state(state);

    Ok(router.into())
}
