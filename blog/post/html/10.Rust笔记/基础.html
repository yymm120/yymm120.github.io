                                                                                                                                                                      2.1. 所有权原则

                                                                                                                                                                      所有权的转移

                                                                                                                                                                      当变量离开作用域后, Rust会自动调用drop函数清理堆内存.

                                                                                                                                                                      对于堆上的数据, 通常会有一个或多个指针指向它. 如果有s1和s2两个变量同时离开作用域就会发生问题, 因此所有权的概念就诞生了.

                                                                                                                                                                      所有权转移(移动)和浅拷贝的区别:

                                                                                                                                                                      所有权转移后, 原来的变量将失效.

                                                                                                                                                                      let s1 = s2;
// 其他语言, 这个赋值操作是浅拷贝
// rust语言, 这个操作叫做移动.

                                                                                                                                                                      克隆(浅拷贝)

                                                                                                                                                                      Copy

                                                                                                                                                                      let x = 5;
let y = x; 
println!("x = {}, y = {}", x, y);

                                                                                                                                                                      这是移动, 实际上这里也做了浅拷贝. 因为对于x整型这样的基本类型原本就是存储在栈上的, 没理由将栈上的值赋给y后再使x的值无效.

                                                                                                                                                                      在 Rust中, 浅拷贝需要借助Copy特征. Copy特征只能浅拷贝基本类型和以基本类型为元素的引用类型.

                                                                                                                                                                      • 所有整数类型，比如 u32
                                                                                                                                                                      • 布尔类型，bool，它的值是 true 和 false
                                                                                                                                                                      • 所有浮点数类型，比如 f64
                                                                                                                                                                      • 字符类型，char
                                                                                                                                                                      • 元组，当且仅当其包含的类型也都是 Copy 的时候。比如，(i32, i32) 是 Copy 的，但 (i32, String) 就不是
                                                                                                                                                                      • 不可变引用 &T ，例如转移所有权中的最后一个例子，但是注意: 可变引用 &mut T 是不可以 Copy的

                                                                                                                                                                      引用

                                                                                                                                                                      fn main() {
    let x = 5;

}

                                                                                                                                                                      解引用

                                                                                                                                                                      • *y
                                                                                                                                                                      fn main() {
    let x = 5;

    assert_eq!(5, x);

}

                                                                                                                                                                      引用规则总结

                                                                                                                                                                      • 引其值, 令可变, 再引其值, 令可变, 同域, 译错.
                                                                                                                                                                      • 引其值, 零可变, 再引其值, 令可变, 异域, 过.
                                                                                                                                                                      • 引其值, 令可变, 再引其值, 不可变, 同域, 译错.
                                                                                                                                                                      • 引其值, 不可变, 再引其值, 不可变, 同域, 过.
                                                                                                                                                                      • 引其值, 令可变, 不可再引, 消其引, 再引其值, 过.
                                                                                                                                                                          let mut s = String::from("hello");

    let r1 = &s; // 没问题
    let r2 = &s; // 没问题
    println!("{} and {}", r1, r2);
    // 此位置之后 r1 和 r2 不再使用

    let r3 = &mut s; // 没问题
    println!("{}", r3);

                                                                                                                                                                      宏

                                                                                                                                                                      从根本上来说，宏是通过一种代码来生成另一种代码，如果大家熟悉元编程，就会发现两者的共同点。

                                                                                                                                                                      在 Rust 中宏分为两大类：声明式宏( *declarative macros* ) macro_rules! 和三种过程宏( *procedural macros* ):

                                                                                                                                                                      Rust 内置的宏都有自己约定俗成的使用方式，例如 vec![...]、assert_eq!(...) 等。

                                                                                                                                                                      宏的作用：

                                                                                                                                                                      1. DRY （Dont Repeat Yourself）不写重复代码

                                                                                                                                                                      2. DSL 领域专用语言

                                                                                                                                                                      3. 可变参数接口, 例如println!就可以传递多个参数。

                                                                                                                                                                      创建声明式宏 macro_rules!

                                                                                                                                                                      最广泛的宏就是声明式宏。

                                                                                                                                                                      macro_rules! say_hello {
	() => (
        println!("hello!");
    )
}
fn main() {
    say_hello!(); // 等同于`println!("hello!");
}

                                                                                                                                                                      extern crate proc_macro;

use proc_macro::TokenStream;
use quote::quote;
use syn;
use syn::DeriveInput;

#[proc_macro_derive(HelloMacro)]
pub fn hello_macro_derive(input: TokenStream) -> TokenStream {
    // 基于 input 构建 AST 语法树
    let ast:DeriveInput = syn::parse(input).unwrap();

    // 构建特征实现代码
    impl_hello_macro(&ast)
}

fn impl_hello_macro(ast: &syn::DeriveInput) -> TokenStream {
    let name = &ast.ident;
    let gen = quote! {
        impl HelloMacro for #name {
            fn hello_macro() {
                println!("Hello, Macro! My name is {}!", stringify!(#name));
            }
        }
    };
    gen.into()
}

                                                                                                                                                                      创建过程宏 - 类属性宏 proc_macro_attribute

                                                                                                                                                                      derive 只能用于结构体和枚举，而类属性宏可以用于其它类型项，例如函数。

                                                                                                                                                                      使用示例：

                                                                                                                                                                      #[route(GET, "/")]
fn index() {}

                                                                                                                                                                      这个#[route]属性就是一个过程宏

                                                                                                                                                                      #[proc_macro_attribute]
pub fn route(attr: TokenStream, item: TokenStream) -> TokenStream {
}

                                                                                                                                                                      与derive宏不同，类属性宏有两个参数。以#[route]宏举例：

                                                                                                                                                                      • 第一个参数是 GET, "/" （所有参数）。
                                                                                                                                                                      • 第二个参数是fn index() {...}这个rust函数源码本身。

                                                                                                                                                                      并且都是TokenStream类型。

                                                                                                                                                                      除了参数有区别之外，它和derive宏的创建没有区别。

                                                                                                                                                                      创建过程宏 - 类函数宏 proc_macro

                                                                                                                                                                      类函数宏与声明宏macro_rules类似。

                                                                                                                                                                      区别在于macro_rules的定义形式与match匹配很像，而类函数宏的定义形式使用的是过程宏的定义方式。

                                                                                                                                                                      #[proc_macro]
pub fn sql(input: TokenStream) -> TokenStream {}

                                                                                                                                                                      而使用形式则类似于函数调用。

                                                                                                                                                                      let sql = sql!(SELECT * FROM posts WHERE id=1);

                                                                                                                                                                      为什么不直接使用声明宏macro_rules来定义呢？

                                                                                                                                                                      因为这里需要对SQL语句进行解析并检查其正确性，这个复杂过程是macro_rules很难做到的。

                                                                                                                                                                      宏 - 重载

                                                                                                                                                                      宏可以重载。

                                                                                                                                                                      macro_rules! test {
    ($left:expr; and $right:expr) => (
        println!("{:?} and {:?} is {:?}", stringify($left), stringify($right), $left && $right)
    );
    ($left:expr; or $right:expr) => (
        println!("{:?} and {:?} is {:?}", stringify($left), stringify($right), $left && $right)
    );
}

test!(1i32 + 1 == 2i32; and 2i32 * 2 == 4i32);
test!(true; or false);

                                                                                                                                                                      宏 - 重复

                                                                                                                                                                      参数列表可以使用 +或者*来表示一个参数出现一次或多次。

                                                                                                                                                                      例如：$(...),+可以匹配一个货多个用逗号分割的表达式。

                                                                                                                                                                      macro_rules! find_min {
    ($x:expr) => ($x);
    // `$x`后面至少跟着一个`$y`
    ($x:expr, $($y:expr),+) => (
        std::comp::min($x, find_min!($($y),+))
    )
}

find_min!(1u32); // 1
find_min!(1u32, 2u32); // 1
find_min!(1u32, 2u32, 3u32); // 1
