                                                                                                                                                                      向上和向下转型

                                                                                                                                                                      • 向上转型目的：简化代码

                                                                                                                                                                      • 特点1：类型是父类，但实际是子类，优先使用的是子类的方法。

                                                                                                                                                                      • 特点2：向上转型后，只能使用重写（父子共有）的方法，不能使用独有的方法。

                                                                                                                                                                      • 向下转型目的：是为了使用子类的方法

                                                                                                                                                                      • 向下转型一般在向上转型之后，再转回来。

                                                                                                                                                                      •   // 向上转型
  Object obj = new String();
  // 向下转型
  String str = (String)obj;

                                                                                                                                                                      static

                                                                                                                                                                      • static修饰的变量和方法叫类成员，随着类的加载而分配内存空间。

                                                                                                                                                                      • 静态不能访问非静态。

                                                                                                                                                                      • 非静态的变量叫成员变量，在实例化后分配内存空间。

                                                                                                                                                                      final

                                                                                                                                                                      • final类不能被继承
                                                                                                                                                                      • final变量必须赋初值
                                                                                                                                                                      • final方法不能被子类重写

                                                                                                                                                                      重载（静态多态）overload

                                                                                                                                                                      • 通过参数的三个不同来实现重载（类型、数量、顺序）
                                                                                                                                                                      • 返回值不同，是无法重写的。
                                                                                                                                                                      • 在创建和使用的时候，知道它使用的哪个方法，这是一种静态多态。

                                                                                                                                                                      重写（动态多态）override

                                                                                                                                                                      • 重写是子类对父类方法的重写。（override）
                                                                                                                                                                      • 重写的条件一是方法声明完全相同（返回值、方法名、参数）。二是修饰符不得低于父类方法的修饰符。
                                                                                                                                                                      • 动态多态：在运行过程中，子类没有就去父类找。这是一种动态多态。

                                                                                                                                                                      Shop.class

                                                                                                                                                                      class Shop implement comAPI{
    // 第二维度表示添加商品的name, price, stock库存。
    private static Object[][] coms = new Object[100][]; // 用于创建商品
    private static int count = 0;

    public Shop() {
    }

    public Object[] getComs() {
        return coms;
    }

    /**
     * creat()
     * @param objs: {name, price, num} 商品名,价格，数量
     */
    @Override
    public void creat(Object[] objs) {
        coms[count] = new Object[]{objs[0], objs[1], objs[2]};
        count++;
    }

    /**
     * append():添加商品库存
     * @param objs：{name, num}要添加的数量。
     */
    @Override
    public void append(Object[] objs) {
        // 解析objs
        String name = (String) objs[0];
        int num = (int)objs[1];
        // 循环查找name的商品
        for (int i = 0; i < count; i++) {
            if (name.equals((String) coms[i][0])){
                coms[i][2] = (int)coms[i][2] + num;
            }
        }
    }

    /**
     * traverse：遍历现有的商品的信息
     * @param objs
     */
    @Override
    public void traverse(Object[] objs) {
        System.out.println("==========Shop商品库存信息遍历=========");
        for (int i = 0; i < count; i++) {
            System.out.println((String)coms[i][0] + "\t 单价："
                    + (Double)coms[i][1] + "\t 库存：" + (Integer)coms[i][2]);
        }
        System.out.println("=====================================");
    }
}

                                                                                                                                                                      @Override
public void showMenu(Object[] objs) {
    System.out.println("【1】creat创建商品");
    System.out.println("【2】append添加商品库存");
    System.out.println("【3】traverse遍历商品信息");
    System.out.println("【4】quit退出");
    System.out.print("选择：");
}

@Override
public void select(Object[] objs) {
    // 参数解析
    int n = (int)objs[0];
    Scanner sc = (Scanner)objs[1];
    switch (n){
        case 1:					// 【1】creat()创建商品
            System.out.print("输入商品名称、价格、数量\n");
            objs = new Object[]{// 构建参数{name, price, num}
                    sc.next(), sc.nextDouble(), sc.nextInt(), sc};
            creat(objs);		// 调用creat()创建商品
            break;
        case 2:					// 【2】append()添加库存
            System.out.println("输入要添加的商品和数量：");
            objs = new Object[]{// 构建参数{name, num}
                sc.next(), sc.nextInt(), sc};
            append(objs);		// 调用append()添加库存
            break;
        case 3:					// 【3】打印商店库存信息
            traverse(objs);
            break;
        default:				// 【4】quit退出
            System.exit(0);
    }
}

                                                                                                                                                                      • HashMap 与 Hashtable的区别
                                                                                                                                                                      1. 都是键值对集合(key, value)， 在key相同时，新值覆盖旧值。
                                                                                                                                                                      2. HashMap是JDK1.2出现的，Hashtable是JDK1.0出现的。
                                                                                                                                                                      3. HashMap是线程异步的，效率高。Hashtable是线程同步的，效率较低。
                                                                                                                                                                      4. HashMap可以存放null键和null值，Hashtable不能存放null。
