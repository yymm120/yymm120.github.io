                                                                                                                                                                      遍历文件

                                                                                                                                                                      案例：复制指定文件夹内的.java文件到指定目录。

                                                                                                                                                                      1. 遍历所有子文件
                                                                                                                                                                      // 1. 定义List存放文件
private final List<File> files = new ArrayList<>();
private List<File> getFile(File f1) {
    // 1. 遍历并筛选
    Stream.of(f1.listFiles()).filter(x->{
        // 2. 如果是文件，则添加到列表
        if (x.isFile()) files.add(x);
        // 3. 筛选，仅保留目录
        return x.isDirectory();
        // 4. 递归
    }).forEach(this::getFile); 
    return files;
}

                                                                                                                                                                      1. 复制所有.java文件到目标路径
                                                                                                                                                                      /**
* 将fs文件列表复制到目录f2中
*/
private void copyFile(List<File> fs, File f2) {
fs.forEach(x->{
    try {
        // 1. 创建目标目录
        if (!f2.exsits()) f2.mkdirs();
        // 2. 创建目标文件
        File file = new File(f2.getPath() + "/" + x.getName() + "副本");
        // 3. 定义输入输出流
        FileInputStream fis = new FileInputStream(x);
        FileOutputStream fos = new FileOutputStream(file);
        // 4. 定义存放字节数组
        byte[] bytes = new byte[(int) x.length()];
        // 5. 执行复制粘贴操作
        fis.read(bytes);
        fos.write(bytes);
        // 关闭资源
        fis.close();
        fos.close();
    } catch (IOException e) {
        e.printStackTrace();
    }
});
}
